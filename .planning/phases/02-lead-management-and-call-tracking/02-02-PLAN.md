    **useAppResume.ts:**
    ```typescript
    import { App } from "@capacitor/app"
    import { useEffect, useRef } from "react"

    export function useAppResume(onResume: () => void) {
      const onResumeRef = useRef(onResume)
      onResumeRef.current = onResume

      useEffect(() => {
        const listenerPromise = App.addListener("appStateChange", ({ isActive }) => {
          if (isActive) {
            onResumeRef.current()
          }
        })
        return () => {
          void listenerPromise.then((handle) => handle.remove())
        }
      }, []) // empty deps — ref keeps callback current without re-registering
    }
    ```

    **useCallTracking.ts:**
    - Props: leadId: string, userId: string
    - State: pendingCall: { initiatedAt: string } | null (useState), showFollowUp: boolean (useState)
    - Also keep a pendingCallRef = useRef(pendingCall) and sync it: pendingCallRef.current = pendingCall — this avoids stale closure in the appStateChange listener
    - const { mutate: logCallEvent } = useCallEventMutation()

    handleCallPress():
    - const initiatedAt = new Date().toISOString()
    - logCallEvent({ lead_id: leadId, user_id: userId, initiated_at: initiatedAt }) fire-and-forget (no await)
    - setPendingCall({ initiatedAt })
    - NOTE: The tel: link is on the anchor element; do NOT use window.location.href here

    useAppResume callback:
    - Check pendingCallRef.current (not pendingCall state — avoids stale closure)
    - If non-null: setShowFollowUp(true)

    handleFollowUpOutcome(outcome: CallOutcome | "cancelled"):
    - If !pendingCall: return
    - const returnedAt = new Date().toISOString()
    - const durationSec = Math.round((new Date(returnedAt).getTime() - new Date(pendingCall.initiatedAt).getTime()) / 1000)
    - logCallEvent({ lead_id: leadId, user_id: userId, initiated_at: pendingCall.initiatedAt, returned_at: returnedAt, outcome: outcome === "cancelled" ? null : outcome, duration_sec: durationSec })
    - setPendingCall(null)
    - setShowFollowUp(false)

    Return: { handleCallPress, showFollowUp, handleFollowUpOutcome }

    Import CallOutcome from @/types; useAppResume from ./useAppResume; useCallEventMutation from ./useLeadMutations.
